<?php
defined('_JEXEC') or die('Restricted access');

use Joomla\CMS\Factory;
use Joomla\CMS\Session\Session;
use Joomla\Database\DatabaseInterface;

/* -----------------------------------------------------
   1) Load Lotteries from the known absolute path
   ----------------------------------------------------- */
$jsonFile = '/home/oscara/web/lottoexpert.net/public_html/lotteries.json';

// Debug check: does the file exist?
if (!file_exists($jsonFile)) {
    echo "<p style='color:red'>Error: File does not exist at $jsonFile</p>";
    return;
}

// Attempt to read the file
$jsonData = file_get_contents($jsonFile);
if ($jsonData === false) {
    echo "<p style='color:red'>Error: Could not read $jsonFile (check permissions)</p>";
    return;
}

// Decode the JSON
$data = json_decode($jsonData, true);
if (json_last_error() !== JSON_ERROR_NONE) {
    echo "<p style='color:red'>Error decoding JSON: " . json_last_error_msg() . "</p>";
    return;
}

// Access the 'lotteries' array
if (!isset($data['lotteries']) || !is_array($data['lotteries'])) {
    echo "<p style='color:red'>Error: JSON does not have a valid 'lotteries' key.</p>";
    return;
}

$lotteries = $data['lotteries'];

/* -----------------------------------------------------
   1b) SKAI Lab – handle "Save best settings" POST
   ----------------------------------------------------- */
$app   = Factory::getApplication();
$input = $app->getInput();

// Derive a lottery_id for saving settings.
// Prefer gameId / game_id from URL; fall back to generic label.
$lotteryId = (string) $input->getString(
    'gameId',
    $input->getString('game_id', '')
);

// Optional: you can later refine this to map from lotteryIndex → $lotteries[...] if needed.

if ($input->getMethod() === 'POST' && $input->getString('save_skai_best_settings', '') !== '') {
    // CSRF check (Joomla form token)
    if (!Session::checkToken()) {
        $app->enqueueMessage('Invalid session token while saving SKAI settings.', 'error');
    } else {
        $user = Factory::getUser();
        if ($user->guest) {
            $app->enqueueMessage('Please log in to save SKAI settings.', 'warning');
        } else {
            /** @var DatabaseInterface $db */
            $db = Factory::getContainer()->get(DatabaseInterface::class);

            // Setting label from form; fallback if empty
            $settingName = trim((string) $input->getString('setting_name', ''));
            if ($settingName === '') {
                $settingName = 'SKAI Lab – ' . date('Y-m-d');
            }

            // Raw JSON string from hidden field; we will validate it
            $paramsJson = (string) $input->getRaw('params', '');
            $decoded    = json_decode($paramsJson, true);

            if (!is_array($decoded)) {
                $app->enqueueMessage('Could not save SKAI settings: invalid parameter payload.', 'error');
            } else {
                // Re-encode to ensure clean JSON
                $cleanJson = json_encode($decoded, JSON_UNESCAPED_SLASHES | JSON_UNESCAPED_UNICODE);

                try {
                    $query = $db->getQuery(true)
                        ->insert($db->quoteName('#__user_saved_settings'))
                        ->columns([
                            $db->quoteName('user_id'),
                            $db->quoteName('lottery_id'),
                            $db->quoteName('source'),
                            $db->quoteName('setting_name'),
                            $db->quoteName('params'),
                            $db->quoteName('created_at'),
                        ])
                        ->values(implode(',', [
                            (int) $user->id,
                            $db->quote($lotteryId),
                            $db->quote('SKAI_LAB'), // source identifier for this lab
                            $db->quote($settingName),
                            $db->quote($cleanJson),
                            $db->quote(date('Y-m-d H:i:s')),
                        ]));

                    $db->setQuery($query);
                    $db->execute();

                    $app->enqueueMessage('SKAI best settings saved for future use.', 'message');
                } catch (\Throwable $e) {
                    $app->enqueueMessage('Error saving SKAI settings: ' . $e->getMessage(), 'error');
                }
            }
        }
    }
}

/* -----------------------------------------------------
   2) Normalize Lottery State Names & Labels
   ----------------------------------------------------- */
foreach ($lotteries as &$lot) {
    // Normalize state field (handle alternative keys, trim spaces)
    $stateKey = 'state';
    if (!isset($lot[$stateKey])) {
        foreach (['State', 'STATE', ' state', 'state '] as $altKey) {
            if (isset($lot[$altKey])) {
                $stateKey = $altKey;
                break;
            }
        }
    }
    $stateValue = isset($lot[$stateKey]) ? trim($lot[$stateKey]) : '';
    // Force Multi-State for specific game IDs regardless of existing state value
    if (in_array($lot['gameId'], ['101','113','128','132','134','135'])) {
        $stateValue = 'Multi-State';
    } elseif ($stateValue === '') {
        $stateValue = 'Unknown';
    }
    $lot['state'] = $stateValue;
    $lot['label'] = $stateValue . ' - ' . $lot['label'];
}
unset($lot);


// Optionally sort the lotteries by label
usort($lotteries, function($a, $b) {
    return strcmp($a['label'], $b['label']);
});

/* -----------------------------------------------------
   3) Determine Which Lottery Was Chosen by GET param
   ----------------------------------------------------- */
$app   = Factory::getApplication();
$input = $app->getInput();

// Use Joomla input filtering and clamp to valid range
$chosenLotteryIndex = $input->getInt('lotteryIndex', 0);
if ($chosenLotteryIndex < 0 || $chosenLotteryIndex >= count($lotteries)) {
    $chosenLotteryIndex = 0;
}
$chosen = $lotteries[$chosenLotteryIndex];

/* -----------------------------------------------------
   4) Fetch DB Data for the Chosen Lottery
   ----------------------------------------------------- */
$db = Factory::getDbo();
$tableName = $chosen['tableName'];
$gameId    = $chosen['gameId'];

$query = $db->getQuery(true)
    ->select('*')
    ->from($db->quoteName($tableName))
    ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
    ->order($db->quoteName('draw_date') . ' ASC');
$db->setQuery($query);
$historicalResults = $db->loadObjectList();

$draws = [];
$lotteryName = $chosen['label'];
if (!empty($historicalResults) && property_exists($historicalResults[0], 'game_name')) {
    $lotteryName = $historicalResults[0]->game_name;
}
$pickSize = (int) $chosen['pickSize'];
$hasExtra = ((int) $chosen['rangeExtra'] > 0);

foreach ($historicalResults as $row) {
    // Build exactly $pickSize “main” numbers:
    $tempNums = [];
    // Define the sequence of DB fields you might pull
    $fields = ['first','second','third','fourth','fifth','sixth','seventh'];
    for ($i = 0; $i < $pickSize; $i++) {
        $f = $fields[$i] ?? null;
        if ($f && isset($row->$f)) {
            $tempNums[] = (int) $row->$f;
        }
    }
    // If there’s an extra ball, it will be the next field after the mains
    $powerball = null;
    if ($hasExtra) {
        $extraField = $fields[$pickSize] ?? null;
        if ($extraField && isset($row->$extraField)) {
            $powerball = (int) $row->$extraField;
        }
    }
    $draws[] = [
        'numbers'   => $tempNums,
        'powerball' => $powerball,
        'draw_date' => $row->draw_date
    ];
}

$countQuery = $db->getQuery(true)
    ->select('COUNT(*)')
    ->from($db->quoteName($tableName))
    ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId));
$db->setQuery($countQuery);
$totalEntriesInDB = (int)$db->loadResult();


// === NEW helper to compute “Weighted Hit Score” per draw ===
function weightedHitScore(array $actualMain, ?int $actualBonus,
                          array $predMainTop20, ?int $predictedBonus,
                          int $wMain, int $wBonus): int {
    $mainHits = count(array_intersect($actualMain, $predMainTop20));
    $bonusHit = ($actualBonus !== null && $predictedBonus === $actualBonus) ? 1 : 0;
    return $wMain * $mainHits + $wBonus * $bonusHit;
}

          [[!-- Epoch Range Explanation --]]
          [[details]]
            [[summary]][[strong]]Epoch Range[[/strong]][[/summary]]
           [[p]]
The epoch range determines how many times the entire dataset is passed through the model during the training process. Training a model involves adjusting its internal parameters (weights) to learn patterns from data and make accurate predictions. Each time the model sees the dataset and updates its parameters, it completes one "epoch."
[[/p]]
[[ul]]
  [[li]]
    [[strong]]What is an epoch?[[/strong]] An epoch represents one full pass of the training dataset through the model. During each pass:
    [[ul]]

// ---- helper: compute a deterministic run signature from settings ----
function computeRunSignature(settingsSnapshot) {
  var keys = Object.keys(settingsSnapshot).sort();
  var str  = keys.map(function(k){ return k + '=' + settingsSnapshot[k]; }).join('|');
  var h = 0;
  for (var i = 0; i < str.length; i++) {
    h = ((h << 5) - h + str.charCodeAt(i)) | 0;
  }
  return 'sig-' + Math.abs(h).toString(16).toUpperCase().padStart(8,'0');
}

function computeSensitivityReport(results) {
  // results: array of topSettings objects (each has all setting fields + compositeScore)
  if (!results || results.length < 2) { return []; }

  var settings = [
    { key:'epochs',         label:'Epochs' },
    { key:'batchSize',      label:'Batch Size' },
    { key:'dropoutRate',    label:'Dropout Rate' },
    { key:'learningRate',   label:'Learning Rate' },
    { key:'activationFunction', label:'Activation Fn' },
    { key:'numHiddenLayers', label:'Hidden Layers' },
    { key:'recencyDecay',   label:'Recency Decay' },
    { key:'skipAlpha',      label:'Skip Alpha' },
    { key:'skaiFreqWeight', label:'Main Freq Wt' },
    { key:'skaiSkipWeight', label:'Main Skip Wt' },
    { key:'skaiHistWeight', label:'Main Hist Wt' },
    { key:'skaiBlendSkipPct', label:'Blend Skip %' },
    { key:'skaiSamplingTemp', label:'Temperature' },
    { key:'skaiDiversityPen', label:'Diversity Pen' },
    { key:'skaiGapScale',   label:'Gap Scale' },
    { key:'laplaceK',       label:'Laplace K' },
    { key:'skipGamma',      label:'Skip Gamma' },
    { key:'strategy',       label:'Strategy' },
    { key:'riskProfile',    label:'Risk Profile' }
  ];

  var report = [];
  for (var si = 0; si < settings.length; si++) {
    var sd = settings[si];
    var key = sd.key;

    // Group results by distinct value of this setting
    var groups = {};
    for (var ri = 0; ri < results.length; ri++) {
      var r = results[ri];
      var val = r[key];
      if (val === undefined || val === null) { continue; }
      var vStr = String(val);
      if (!groups[vStr]) { groups[vStr] = []; }
      groups[vStr].push(r.compositeScore);
    }

    var groupKeys = Object.keys(groups);
    if (groupKeys.length < 2) { continue; } // not enough variation to measure impact

    // Compute mean score per value group
    var groupMeans = {};
    for (var g = 0; g < groupKeys.length; g++) {
      var gk = groupKeys[g];
      var arr = groups[gk];
      var sum = 0;
      for (var ai = 0; ai < arr.length; ai++) { sum += arr[ai]; }
      groupMeans[gk] = sum / arr.length;
    }

    // Overall mean score
    var allScores = results.map(function(r){ return r.compositeScore; });
    var overallMean = allScores.reduce(function(a,b){ return a+b; }, 0) / allScores.length;

    // Impact = max deviation of group mean from overall mean
    var maxImpact = 0;
    var bestVal = '';
    var direction = 0;
    for (var g = 0; g < groupKeys.length; g++) {
      var gk = groupKeys[g];
      var dev = groupMeans[gk] - overallMean;
      if (Math.abs(dev) > Math.abs(maxImpact)) {
        maxImpact = dev;
        bestVal = gk;
      }
    }
    direction = maxImpact > 0 ? '+' : (maxImpact < 0 ? '−' : '0');

    // Variance across group means
    var allMeans = groupKeys.map(function(gk){ return groupMeans[gk]; });
    var meanOfMeans = allMeans.reduce(function(a,b){ return a+b; }, 0) / allMeans.length;
    var variance = allMeans.reduce(function(acc, m){ return acc + Math.pow(m - meanOfMeans, 2); }, 0) / allMeans.length;

    report.push({
      label:       sd.label,
      valuesTested: groupKeys.length,
      impact:      maxImpact,
      variance:    variance,
      direction:   direction,
      bestVal:     bestVal
    });
  }

  // Sort by absolute impact descending
  report.sort(function(a, b){ return Math.abs(b.impact) - Math.abs(a.impact); });
  return report;
}


// 1) Table Sort
function sortTable(tableId, columnIndex){
  const table = document.getElementById(tableId);
  if(!table)return;
  const tbody = table.querySelector("tbody");
  const rows = Array.from(tbody.querySelectorAll("tr"));
  let asc = (table.getAttribute("data-sortcol") == columnIndex && table.getAttribute("data-sortdir")=="asc") ? false : true;
  rows.sort((a, b) => {
    const aText = a.cells[columnIndex].innerText.trim();
    const bText = b.cells[columnIndex].innerText.trim();
    const aNum = parseFloat(aText.replace(/[^\d.-]/g,''));
    const bNum = parseFloat(bText.replace(/[^\d.-]/g,''));
    if(!isNaN(aNum) && !isNaN(bNum))
      return asc ? aNum - bNum : bNum - aNum;
    else
      return asc ? aText.localeCompare(bText) : bText.localeCompare(aText);
  });
  rows.forEach(r => tbody.removeChild(r));
  rows.forEach(r => tbody.appendChild(r));
  table.setAttribute("data-sortcol", columnIndex);
  table.setAttribute("data-sortdir", asc ? "asc" : "desc");
}

// 2) customBinaryCrossentropy + reduceLROnPlateau
function customBinaryCrossentropy(yTrue, yPred){
  const EPS = 1e-7;
  return tf.tidy(()=>{
    const clipped = yPred.clipByValue(EPS, 1-EPS);
    const term1 = yTrue.mul(clipped.log()).neg();
    const term2 = tf.scalar(1).sub(yTrue).mul(tf.scalar(1).sub(clipped).log()).neg();
    return term1.add(term2).mean();
  });
}
function reduceLROnPlateau(opt, factor=0.5, patience=4, minLr=1e-7){
  let bestVal = Infinity, wait = 0;
  return {
    onEpochEnd: async (ep, logs) => {
      const valLoss = logs.val_loss;
      if(valLoss < bestVal){
        bestVal = valLoss; wait = 0;
      } else {
        wait++;
        if(wait >= patience){
          let lr;
          if(typeof opt.learningRate === 'object'){
            lr = opt.learningRate.dataSync()[0];
          } else {
            lr = opt.learningRate;
          }
          let newLr = lr * factor; if(newLr < minLr)newLr = minLr;
          if(typeof opt.learningRate === 'object'){
            opt.learningRate.assign(newLr);
          } else {
            opt.learningRate = newLr;
          }
          console.log(`ReduceLROnPlateau => epoch ${ep}, new lr: ${newLr}`);
          wait = 0;
        }
      }
    }
  };
}

// 3) For epoch-end distribution updates
function computeEpochDist(predArr, valArr, N){
  const topK = 20;
  const dist = [0,0,0,0,0,0];
  for(let i = 0; i < valArr.length; i++){
    const mainProbs = predArr[i].slice(0, N);
    const sorted = mainProbs.map((v, idx) => ({ v, index: idx+1 }));
    sorted.sort((a, b) => b.v - a.v);
    const topIdx = sorted.slice(0, topK).map(o => o.index);
    const actual = [];
    for(let j = 0; j < N; j++){
      if(valArr[i][j] === 1) actual.push(j+1);
    }
    const c = actual.filter(num => topIdx.includes(num)).length;
    if(c >= 0 && c < dist.length){
      dist[c]++;
    }
  }
  return dist;
}

  // NEW: compute per-draw Skip feature vectors (normalized 0..1) for each main number.
  // drawsNumbers: [ [n1, n2, ...], ... ] in chronological order (oldest → newest).
  // Returns: skipsPerStep[t][k] = normalized "draws since last hit" for number (k+1) at draw t.
  function computeSkipFeatureSeries(drawsNumbers, pickSize, mainNumbersMax) {
    const lastHitIndex = new Array(mainNumbersMax).fill(null);
    const skipsPerStep = [];

    for (let t = 0; t < drawsNumbers.length; t++) {
      const skipVec = new Array(mainNumbersMax).fill(0);

      // For each number, how long since it last appeared?
      for (let k = 0; k < mainNumbersMax; k++) {
        const last = lastHitIndex[k];
        const skip = (last === null) ? t : (t - last - 1);
        skipVec[k] = skip;
      }

      // Update lastHitIndex using the CURRENT draw's main numbers.
      const draw = drawsNumbers[t];
      for (let j = 0; j < pickSize; j++) {
        const n = draw[j];
        if (n >= 1 && n <= mainNumbersMax) {
          lastHitIndex[n - 1] = t;
        }
      }

      skipsPerStep.push(skipVec);
    }

    // Normalize skips to 0..1 so they combine safely with other features.
    let maxSkip = 0;
    for (let t = 0; t < skipsPerStep.length; t++) {
      const vec = skipsPerStep[t];
      for (let i = 0; i < vec.length; i++) {
        if (vec[i] > maxSkip) {
          maxSkip = vec[i];
        }
      }
    }

    if (maxSkip > 0) {
      for (let t = 0; t < skipsPerStep.length; t++) {
        const vec = skipsPerStep[t];
        for (let i = 0; i < vec.length; i++) {
          vec[i] = vec[i] / maxSkip;
        }
      }
    }

    return skipsPerStep;
  }

  function evaluateThresholds(predArr, valArr, thresholds){
    let bestF1 = -Infinity, bestThr = 0.5;
    for(let thr of thresholds){
      let tp = 0, fp = 0, fn = 0;
      for(let i = 0; i < valArr.length; i++){
        for(let j = 0; j < N; j++){
          const predicted = predArr[i][j] > thr ? 1 : 0;
          const actual = valArr[i][j];
          if(predicted === 1 && actual === 1) tp++;
          else if(predicted === 1 && actual === 0) fp++;
          else if(predicted === 0 && actual === 1) fn++;
        }
      }
      const prec = (tp + fp) === 0 ? 0 : tp / (tp + fp);
      const rec = (tp + fn) === 0 ? 0 : tp / (tp + fn);
      const f1 = (prec + rec) === 0 ? 0 : 2 * (prec * rec) / (prec + rec);
      if(f1 > bestF1){
        bestF1 = f1; bestThr = thr;
      }
    }
    return { bestF1, bestThr };
  }

    function parseStepRange(str, step){
      step = step || 1;

        return [];

      if (str.indexOf('-') === -1) {
        var vSingle = parseFloat(str);
        return isNaN(vSingle) ? [] : [vSingle];

      // Range "a-b" → build a, a+step, …, b (clamped properly)
      var parts = str.split('-');
      var lo = parseFloat(parts[0]);
      var hi = parseFloat(parts[1]);
      if (isNaN(lo) || isNaN(hi)) {
        return [];
      }
      if (lo > hi) {
        var tmp = lo;
        lo = hi;
        hi = tmp;
      }

      var out = [];
      if (step <= 0) {
        step = 1;
      }

      // Start at the first multiple of `step` ≥ lo.
      var start = Math.ceil(lo / step) * step;
      for (var v = start; v <= hi; v += step) {
        out.push(v);
      }
      return out;

                  varF1_tail = foldF1s.reduce(function (acc, val) {
                    return acc + Math.pow(val - avgF1_tail, 2);
                  }, 0) / foldF1s.length;
                }
                const stdF1_tail = Math.sqrt(varF1_tail);

                // === Aggregate metrics for this setting (windows + draws) ===
                const usedWindows = foldF1s.length || 1; // actual number of executed windows
                const avgF1 = totalF1 / usedWindows;
                const varF1 = foldF1s.length
                  ? foldF1s.reduce(function (acc, val) {
                      return acc + Math.pow(val - avgF1, 2);
                    }, 0) / foldF1s.length
                  : 0;
                const stdF1 = Math.sqrt(varF1);

                // Use totalValExamples as denominator for draw-based metrics
                const denomDraws = totalValExamples || 1;
                const avgMatches = denomDraws ? matchCount / denomDraws : 0;
                const avgAcc = usedWindows ? totalAcc / usedWindows : 0;
                const avgLoss = usedWindows ? totalLoss / usedWindows : 0;
                const avgPrec = usedWindows ? totalPrec / usedWindows : 0;
                const avgRec = usedWindows ? totalRec / usedWindows : 0;
                const avgRank = denomDraws ? bestRankSum / denomDraws : 0;

                // perDrawWHS was pushed on each draw inside the fold loop
                const muWHS = perDrawWHS.length
                  ? perDrawWHS.reduce(function (sum, x) { return sum + x; }, 0) / perDrawWHS.length
                  : 0;
                const sigmaWHS = perDrawWHS.length
                  ? Math.sqrt(
                      perDrawWHS.reduce(function (acc, x) {
                        return acc + Math.pow(x - muWHS, 2);
                      }, 0) / perDrawWHS.length
                    )
                  : 0;

                // Tail probabilities:
                // p3 = P(hits ≥ 3), p4 = P(hits ≥ 4), pBonus = P(bonus hit)
                const p3 = denomDraws ? drawsWith3Plus / denomDraws : 0;
                const p4 = denomDraws ? drawsWith4Plus / denomDraws : 0;
                const pBonus = denomDraws ? drawsWithBonusHit / denomDraws : 0;

                // Final selection score using SKAI-style main weights + profile-dependent penalties
                // Normalize freq/skip/hist weights so they sum to 1
                var sumMainW = freqWeight + skipWeight + histWeight;
                if (!sumMainW || !isFinite(sumMainW)) {
                  sumMainW = 1;
                }
                var wf = freqWeight / sumMainW;
                var ws = skipWeight / sumMainW;
                var wh = histWeight / sumMainW;

                // Map lab metrics to SKAI concepts:
                // - metricFreq  => more correct hits (avg WHS)
                // - metricSkip  => stronger tails (p(3+), p(4+))
                // - metricHist  => greater stability across windows
                var metricFreq  = muWHS;
                var metricSkip  = (p3 + p4) / 2;
                var metricHist  = 1 / (1 + stdF1); // higher stdF1 => lower stability

                var skaiWeightedScore =
                  wf * metricFreq +
                  ws * metricSkip +
                  wh * metricHist;

                // Combine SKAI-weighted score with volatility penalty and bonus-tail reward

                  skaiWeightedScore
                  - lambda * sigmaWHS

function applySkaiPreset(mode) {
  // Resolve SKAI fields (may not all exist on every game)
  var elWindow    = document.getElementById('skaiWindowSize');
  var elSkipWin   = document.getElementById('skaiSkipWindow');
  var elFreq      = document.getElementById('skaiFreqWeight');

  var elBlendAi   = document.getElementById('skaiBlendAiPct');
  var elTemp      = document.getElementById('skaiSamplingTemperature');
  var elDiv       = document.getElementById('skaiDiversityPenalty');
  var elGap       = document.getElementById('skaiGapScale');
  var elTopNNums  = document.getElementById('skaiTopNNumbers');
  var elTopNComb  = document.getElementById('skaiTopNCombos');
  var elMode      = document.getElementById('skaiRunMode');

  var cfg;

  // NEW: support both string modes ("balanced", "conservative", "exploratory")
  // and object-based presets passed by the Starter buttons.
  if (mode && typeof mode === 'object') {
    // Map the Starter object shape into the internal cfg structure.
    cfg = {
      windowDraws: (typeof mode.window === 'number' ? mode.window : 600),
      skipWindow:  (typeof mode.skipWindow === 'number' ? mode.skipWindow : 240),
      mainFreq:    (typeof mode.freq === 'number' ? mode.freq : 40),
      mainSkip:    (typeof mode.skip === 'number' ? mode.skip : 40),
      mainHist:    (typeof mode.hist === 'number' ? mode.hist : 20),
      pbFreq:      (typeof mode.pbFreq === 'number' ? mode.pbFreq : 40),
      pbSkip:      (typeof mode.pbSkip === 'number' ? mode.pbSkip : 40),
      pbHist:      (typeof mode.pbHist === 'number' ? mode.pbHist : 20),
      blendSkip:   (typeof mode.blendSkip === 'number' ? mode.blendSkip : 50),

      temp:        (typeof mode.temp === 'number' ? mode.temp : 0.8),
      diversity:   (typeof mode.div === 'number' ? mode.div : 0.10),
      gapScale:    (typeof mode.gap === 'number' ? mode.gap : 1.0),
      topNNumbers: (typeof mode.topNums === 'number' ? mode.topNums : 20),
      topNCombos:  (typeof mode.topCombos === 'number' ? mode.topCombos : 60),
      runMode:     (mode.mode || 'balanced')
    };
  } else {
    // Original behavior: string-based preset selector.
    switch (mode) {
      case 'conservative':
        cfg = {
          windowDraws: 800,
          skipWindow:  300,
          mainFreq:    50,
          mainSkip:    30,
          mainHist:    20,
          pbFreq:      50,
          pbSkip:      30,
          pbHist:      20,
          blendSkip:   30,
          blendAi:     70,
          temp:        0.6,
          diversity:   0.05,
          gapScale:    0.8,
          topNNumbers: 20,
          topNCombos:  50,
          runMode:     'conservative'
        };
        break;

      case 'exploratory':
        cfg = {
          windowDraws: 500,
          skipWindow:  200,
          mainFreq:    30,
          mainSkip:    50,
          mainHist:    20,
          pbFreq:      30,
          pbSkip:      50,
          pbHist:      20,
          blendSkip:   60,
          blendAi:     40,
          temp:        1.0,
          diversity:   0.15,
          gapScale:    1.2,
          topNNumbers: 22,
          topNCombos:  80,
          runMode:     'exploratory'
        };
        break;

      case 'balanced':
      default:
        cfg = {
          windowDraws: 600,
          skipWindow:  240,
          mainFreq:    40,
          mainSkip:    40,
          mainHist:    20,
          pbFreq:      40,
          pbSkip:      40,
          pbHist:      20,
          blendSkip:   50,
          blendAi:     50,
          temp:        0.8,
          diversity:   0.10,
          gapScale:    1.0,
          topNNumbers: 20,
          topNCombos:  60,
          runMode:     'balanced'
        };
        break;
    }
  }

  // Null-safe assignments: only write if the element exists
  if (elWindow)    { elWindow.value    = cfg.windowDraws; }
  if (elSkipWin)   { elSkipWin.value   = cfg.skipWindow; }
  if (elFreq)      { elFreq.value      = cfg.mainFreq; }
  if (elSkip)      { elSkip.value      = cfg.mainSkip; }
  if (elHist)      { elHist.value      = cfg.mainHist; }
  if (elPbFreq)    { elPbFreq.value    = cfg.pbFreq; }
  if (elPbSkip)    { elPbSkip.value    = cfg.pbSkip; }
  if (elPbHist)    { elPbHist.value    = cfg.pbHist; }
  if (elBlendSkip) { elBlendSkip.value = cfg.blendSkip; }
  if (elBlendAi)   { elBlendAi.value   = cfg.blendAi; }
  if (elTemp)      { elTemp.value      = cfg.temp; }
  if (elDiv)       { elDiv.value       = cfg.diversity; }
  if (elGap)       { elGap.value       = cfg.gapScale; }
  if (elTopNNums)  { elTopNNums.value  = cfg.topNNumbers; }
  if (elTopNComb)  { elTopNComb.value  = cfg.topNCombos; }
  if (elMode)      { elMode.value      = cfg.runMode; }
}

  // =====================================================================
  // LIVE COMBINATION COUNT (update whenever inputs change)
  // =====================================================================
  function updateCombosCount() {
    try {
      var eVal2  = (document.getElementById('epochsRange')   || {value:'100'}).value.trim() || '100';
      var bsVal2 = (document.getElementById('batchSizes')    || {value:'4'}).value.trim()   || '4';
      var drVal2 = (document.getElementById('dropoutRates')  || {value:'0.2'}).value.trim() || '0.2';
      var lrVal2 = (document.getElementById('learningRates') || {value:'0.002'}).value.trim() || '0.002';
      var hlVal2 = (document.getElementById('hiddenLayers')  || {value:'2'}).value.trim()   || '2';
      var rdVal2 = (document.getElementById('recencyDecayRates') || {value:'1'}).value.trim() || '1';
      var saVal2 = (document.getElementById('skipAlphaValues') || {value:'0.5'}).value.trim() || '0.5';
      var lkVal2 = (document.getElementById('laplaceK')      || {value:'1'}).value.trim()   || '1';
      var sgVal2 = (document.getElementById('skipGamma')     || {value:'0.2'}).value.trim() || '0.2';
      var rpVal2 = (document.getElementById('riskProfile')   || {value:'balanced'}).value   || 'balanced';
      var stVal2 = (document.getElementById('skaiStrategy')  || {value:'hybrid'}).value     || 'hybrid';
      var exMode = !!(document.getElementById('exhaustiveMode') || {checked:false}).checked;

      function countRange(str) {
        var minV = 5, maxV = 1000;
        str = String(str).trim();
        if (str.indexOf('-') !== -1) {
          var p = str.split('-');
          var lo = parseInt(p[0],10) || minV;
          var hi = parseInt(p[1],10) || lo;
          return Math.max(1, Math.abs(hi - lo) + 1);
        } else {
          return 1;
        }
      }
      function countList(str) {
        return Math.max(1, str.split(',').filter(function(v){ return v.trim() !== ''; }).length);
      }
      function countRangeOrList(str) {
        return str.indexOf('-') !== -1 && str.indexOf(',') === -1 ? countRange(str) : countList(str);
      }

      var nEpoch   = countRange(eVal2);
      var nBatch   = countList(bsVal2);
      var nDrop    = countRangeOrList(drVal2);
      var nLR      = countList(lrVal2);
      var nHidden  = countRangeOrList(hlVal2);
      var nRecency = countList(rdVal2);
      var nSkipA   = countRangeOrList(saVal2);
      var nLaplace = (lkVal2.indexOf(',') !== -1) ? 3 : 1;
      var nSkipG   = (sgVal2.indexOf('-') !== -1 && sgVal2.indexOf(',') === -1) ?
                       countRange(sgVal2) : countList(sgVal2);
      var nRP      = (rpVal2 === 'all') ? 3 : 1;
      var nST      = (stVal2 === 'all') ? 3 : 1;

      var afSel2  = document.getElementById('activationFunctions');
      var nAF     = afSel2 ? Math.max(1, Array.from(afSel2.selectedOptions).length) : 1;

      var total = nEpoch * nBatch * nDrop * nLR * nAF * nHidden * nRecency * nSkipA *
                  nLaplace * nSkipG * nRP * nST;
      if (!exMode && total > 500) { total = 500; }

      var el = document.getElementById('combosCountDisplay');
      var wp = document.getElementById('combosPanel');
      var wm = document.getElementById('combosWarnMsg');
      if (el) { el.textContent = total.toLocaleString(); }
      if (wp) { wp.classList.toggle('warn', total > 400); }
      if (wm) { wm.style.display = (total > 400) ? 'inline' : 'none'; }
    } catch(e) { /* ignore */ }
  }

      function qaPass(id, msgId, msg) {
        var el  = document.getElementById(id);

        if (el)  { el.className = 'qa-pass'; el.textContent = '✔ ' + el.textContent.replace(/^[^\s]+\s/, ''); }

      }
      function qaFail(id, msgId, msg) {
        var el  = document.getElementById(id);
        var mel = document.getElementById(msgId);
        if (el)  { el.className = 'qa-fail'; el.textContent = '✘ ' + el.textContent.replace(/^[^\s]+\s/, ''); }
        if (mel) { mel.textContent = msg; mel.style.color = '#dc2626'; }

      function qaReset(id) {
        var el = document.getElementById(id);
        if (el) { el.className = 'qa-pending'; el.textContent = '⬜ ' + el.textContent.replace(/^[^\s]+\s/, ''); }
      }

        qaReset('qaSmoke');

        var gdSmoke = prepareData();

        if (!gdSmoke.inputs || !gdSmoke.labels) { throw new Error('Invalid dataset tensors'); }
        var nSamples = gdSmoke.numExamples;
        gdSmoke.inputs.dispose(); gdSmoke.labels.dispose();
        qaPass('qaSmoke', 'qaSmokeMsg',
          'Dataset prepared successfully. ' + nSamples + ' draws loaded. Tensors valid.');
      } catch(err) {
        qaFail('qaSmoke', 'qaSmokeMsg', 'Failed: ' + err.message);
      }

      // --- Consistency test: same settings applied twice should give the same combo count ---
      try {
        qaReset('qaConsistency');
        document.getElementById('qaConsistencyMsg').textContent = 'Running...';
        // Check that the registry covers all settings currently in the UI
        var registryKeys = Object.keys(SKAI_SETTINGS_REGISTRY);
        var uiIds = ['epochs','batchSize','dropoutRate','learningRate',
                     'activationFn','numHiddenLayers','recencyDecay','skipAlpha',

                     'topNCombos','laplaceK','skipGamma','riskProfile','strategy'];

          qaFail('qaConsistency', 'qaConsistencyMsg',
            'Registry missing entries for: ' + missing.join(', '));

            'Registry covers all ' + registryKeys.length + ' settings. No gaps found.');
        }
      } catch(err) {
        qaFail('qaConsistency', 'qaConsistencyMsg', 'Failed: ' + err.message);
      }

      // --- Determinism test: same signature from same settings ---

        qaReset('qaDeterminism');

            'Signatures differ: ' + sig1 + ' vs ' + sig2 + '. Non-deterministic.');
        }
      } catch(err) {
        qaFail('qaDeterminism', 'qaDeterminismMsg', 'Failed: ' + err.message);
